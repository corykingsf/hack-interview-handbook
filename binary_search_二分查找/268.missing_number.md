    public int missingNumber(int[] nums) {
        boolean[] visited= new boolean[nums.length + 1];

        for (int num : nums) {
            visited[num] = true;
        }

        for (int i = 0; i < visited.length; i++) {
            if (visited[i] == false) {
                return i;
            }
        }
        return -1;
    }
