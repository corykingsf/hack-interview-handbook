    public int triangleNumber(int[] nums) {
        Arrays.sort(nums);
      // sanity check
        int count = 0;
        for (int k = nums.length - 1; k >=  2; k--) {
            int i = 0; 
            int j = k - 1;
            
            while (i < j) {
                int sum = nums[i] + nums[j];
                if (sum <= nums[k]) {
                    i++;
                } else {
                    count += (j - i);
                    j--;
                }
            }
            
        } 
        return count;
    }
