class LCArray {
    public int[] a;
    public int curSize;
    public int maxSize;


    public LCArray() {
        this.a=new int[1];
        this.curSize=0;
        this.maxSize=1;
    }

    public void push_back(int n) {
        if(curSize==maxSize){
            this.maxSize*=2;
            int[] temp=new int[maxSize];
            for(int i=0;i<curSize;i++)
                temp[i]=a[i];
            this.a=temp;
        }
        this.a[curSize]=n;
        this.curSize+=1;
    }

    public void pop_back() {
        this.curSize-=1;
    }

    public int size() {
        return curSize;
    }

    public int index(int idx) {
        return a[idx];
    }
}
