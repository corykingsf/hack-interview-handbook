class LCArray {
    public int[] a;
    public int curSize;
    public int maxSize;

    public LCArray() {
        this.maxSize = 100;
        this.a = new int[maxSize];
        this.curSize = 0;

    }
    
    public void push_back(int n) {
        if (curSize == maxSize) {
            //double array's size
            int[] tmp = new int[maxSize * 2];
            for (int i = 0; i < curSize; i++) {
                tmp[i] = a[i];
            }
            this.a = tmp;
        }
        this.a[curSize] = n;
        curSize += 1;

    }
    
    public void pop_back() {
        this.curSize--;
    }
    
    public int size() {
        return this.curSize;
    }
    
    public int index(int idx) {
        return a[idx];
    }
}
