class Solution {
    public String longestPalindrome(String s) {
        if (s == null ||s.length() == 0) return "";
        // 中心扩散，枚举中点
        int n = s.length();
        s = " " +s;
        int ansLen = 0;
        int ansStart = 0;  
        // aba   枚举长度为奇数的串
        for (int center = 1; center <= n; center++) {
            int l = center - 1;
            int r = center +1;
            while (l > 0 && r <= n && s.charAt(l) == s.charAt(r)) {
                l--;
                r++;
            }
            // l+1 ~ r -1
            if (r - l - 1 > ansLen) {
                ansLen = r - l-1;
                ansStart = l +1;
            }
            
        }
        
        //abba  枚举长度为偶数的串
        for (int center = 1; center < n; center++) {  //中心是两个字符(一个空档)
            int l = center;
            int r = center +1;
            while (l > 0 && r <= n && s.charAt(l) == s.charAt(r)) {
                l--;
                r++;
            }
            // l+1 ~ r -1
            if (r - l - 1 > ansLen) {
                ansLen = r - l -1;
                ansStart = l +1;
            }
            
        }
        
        return s.substring(ansStart, ansStart + ansLen);
        
    }
}
